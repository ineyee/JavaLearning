一个集合里可以有多个文档，一个文档对应一条数据、数据都以 JSON 的形式存在

## 一、增 insert

* 单条增加

```javascript
db.product.insertOne({
	"name": "iPhoneX",
	"desc": "初代全面屏",
	"price": 8888
});

db.product.insert({
	"name": "iPhoneX",
	"desc": "初代全面屏",
	"price": 8888
});
```

* 批量增加

```javascript
db.product.insertMany([
  { name: "iPhoneX", desc: "初代全面屏", price: 8888 },
  { name: "华为 Mate70", price: 6666 },
  { name: "小米 10", price: 4444 }
]);

db.product.insert([
  { name: "iPhoneX", desc: "初代全面屏", price: 8888 },
  { name: "华为 Mate70", price: 6666 },
  { name: "小米 10", price: 4444 }
]);
```

## 二、删 delete

> db.product.delete({
> 	一个 map 参数，类似于 SQL 语句里的 WHERE，用来写删除条件
> });
>
> 
>
> 删除条件那里
>
> * 只写一个空的 {} 就是：啥条件也没有、全部数据
> * 什么条件符都不写就是：=
> * $ne(not equal)：!=
> * $gt(greater than)：>
> * $gte(greater than equal)：>=
> * $lt(less than)：<
> * $lte(less than equal)：<=
> * $in：IN
>

* 单条删除

```javascript
// 删除 name = iPhoneX 的那条数据
db.product.deleteOne({
	"name": "iPhoneX",
});
```

* 批量删除

```javascript
// MongoDB 的默认主键是 _id，不是 id
db.product.deleteMany({
	"_id": { $in: [ ObjectId("699d4fc4a856defae30dcb4e"), ObjectId("699d4fc4a856defae30dcb4fq	") ] }
});

// 如果自己定义了主键 id
db.product.deleteMany({
  "id": { $in: [ 1, 2, 3 ] }
});
```

* 全部删除

```javascript
// 删除所有数据
db.product.deleteMany({});
```

## 三、改 update

> db.product.update(
> 	{ 第一个 map 参数，类似于 SQL 语句里的 WHERE，用来写修改条件 },
> 	{ 第二个 map 参数，类似于 SQL 语句里的 SET，用来写要修改的字段及新值 }
> );
>
> 
>
> 修改条件那里
> * 只写一个空的 {} 就是：啥条件也没有、全部数据
> * 什么条件符都不写就是：=
> * $ne(not equal)：!=
> * $gt(greater than)：>
> * $gte(greater than equal)：>=
> * $lt(less than)：<
> * $lte(less than equal)：<=
> * $in：IN
>
> 
>
> 要修改的字段及新值那里，固定得写个 $set 的 key**（如果原数据里有这个字段那就会修改，没有这个字段那就会新增这个字段）**

* 单条修改

```javascript
// 找到 name = iPhoneX 的那条数据，把这条数据的 desc 改成 初代全面屏，使用 FaceID
db.product.updateOne(
	{ "name": "iPhoneX" },
	{ $set: { "desc": "初代全面屏，使用 FaceID" } }
);
```

* 批量修改

```javascript
// 找到 price <= 6666 的那些数据，把这些数据的 desc 改成 价格低，质量高
db.product.updateMany(
	{ "price": { $lte: 6666 } },
	{ $set: { "desc": "价格低，质量高" } }
);
```

* 全部修改

```javascript
// 把所有数据的 desc 改成 统统 2 元，先到先得，把所有数据的 price 改成 2 元
db.product.updateMany(
	{},
	{
		$set: {
			"desc": "统统 2 元，先到先得",
			"price": 2
		}
	}
);
```
